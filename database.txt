create table country(
    country_id integer primary key autoincrement,
    country varchar(30));
    
create table language(
    language_id integer primary key autoincrement,
    language varchar(30));
    
create table author(
    author_id integer primary key autoincrement,
    author varchar(50),
    media_type varchar (5));
    
create table genre(
    genre_id integer primary key autoincrement,
    genre varchar(20),
    media_type varchar (5));
    
create table cast(
    cast_id integer primary key autoincrement,
    cast varchar(50));
    
create table movie(
    movie_id integer primary key autoincrement,
    movie_isbn varchar(20),
    movie_title varchar(100) not null,
    movie_cover blob,
    movie_year varchar(10),
    movie_length_minutes varchar(4),
    movie_plot varchar(2000),
    movie_cast varchar(1000),
    movie_author_id int(5),
    movie_genre_id int(4),
    movie_language_id int(4),
    movie_country_id int(4),
    foreign key (movie_author_id)
        references author(author_id),
    foreign key (movie_genre_id)
        references genre(genre_id),
    foreign key (movie_language_id)
        references language(language_id),
    foreign key (movie_country_id)
        references country(country_id));
        
create table book(
    book_id integer primary key autoincrement,
    book_isbn varchar(20),
    book_title varchar(100) not null,
    book_year varchar(10),
    book_plot varchar(2000),
    book_length_pages varchar(8),
    book_cover blob,
    book_genre_id int(4),
    book_author_id int(5),
    foreign key (book_author_id)
        references author(author_id),
    foreign key (book_genre_id)
        references genre(genre_id));
        
create table cd(
    cd_id integer primary key autoincrement,
    cd_isbn varchar(20),
    cd_title varchar(100),
    cd_genre_id int(4),
    cd_cover blob,
    cd_author_id int(5),
    foreign key (cd_author_id)
        references author(author_id),
    foreign key (cd_genre_id)
        references genre(genre_id));
